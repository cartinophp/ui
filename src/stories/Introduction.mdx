## ðŸ”§ Sistema di Themes

I theme sono definiti con `tailwind-variants` in file separati:

**themes/button.js:**
\`\`\`js
import { tv } from 'tailwind-variants'

export default tv({
  slots: {
    root: 'inline-flex items-center justify-center',
    label: 'font-medium'
  },
  variants: {
    size: {
      sm: { root: 'h-8 px-3 text-sm' },
      md: { root: 'h-10 px-4 text-base' },
      lg: { root: 'h-12 px-6 text-lg' }
    },
    variant: {
      solid: { root: 'bg-primary-600 text-white' },
      outline: { root: 'border border-primary-600 text-primary-600' }
    }
  },
  defaultVariants: {
    size: 'md',
    variant: 'solid'
  }
})
\`\`\`

**components/Button.vue:**
\`\`\`vue
<script setup>
import { computed } from 'vue'
import buttonTheme from '../themes/button'

const props = defineProps({
  size: { type: String, default: 'md' },
  variant: { type: String, default: 'solid' }
})

const ui = computed(() => buttonTheme({
  size: props.size,
  variant: props.variant
}))
</script>

<template>
  <button :class="ui.root({ class: props.ui?.root })">
    <span :class="ui.label({ class: props.ui?.label })">
      <slot />
    </span>
  </button>
</template>
\`\`\`
